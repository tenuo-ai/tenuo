1//! Tenuo Authorizer - Minimal Data Plane
2//!
3//! A lightweight, embeddable authorizer that can run as:
4//! - A sidecar container in Kubernetes
5//! - A standalone verification service
6//! - An embedded library (just use the tenuo_core crate directly)
7//!
8//! # Usage
9//!
10//! As a CLI tool:
11//! ```bash
12//! # Verify a warrant
13//! echo $WARRANT | tenuo-authorizer verify --tool upgrade_cluster --arg cluster=staging-web
14//!
15//! # Run as a verification server (for sidecars)
16//! tenuo-authorizer serve --port 9090
17//! ```
18//!
19//! # Kubernetes Sidecar Deployment
20//!
21//! ```yaml
22//! apiVersion: apps/v1
23//! kind: Deployment
24//! spec:
25//!   template:
26//!     spec:
27//!       containers:
28//!       - name: app
29//!         # Your application
30//!       - name: tenuo-authorizer
31//!         image: tenuo/authorizer:latest
32//!         env:
33//!         - name: TENUO_TRUSTED_KEYS
34//!           valueFrom:
35//!             configMapKeyRef:
36//!               name: tenuo-config
37//!               key: trusted_keys
38//!         ports:
39//!         - containerPort: 9090
40//!         resources:
41//!           limits:
42//!             memory: "32Mi"
43//!             cpu: "50m"
44//! ```

46use clap::{Parser, Subcommand};
47use std::collections::HashMap;
48use std::io::{self, Read};
49use std::path::PathBuf;
50use tenuo_core::{
51    constraints::ConstraintValue,
52    planes::Authorizer,
53    revocation::RevocationList,
54    wire, PublicKey,
55};
56
57#[derive(Parser)]
58#[command(name = "tenuo-authorizer")]
59#[command(about = "Tenuo Data Plane Authorizer", long_about = None)]
60struct Cli {
61    /// Trusted public keys (comma-separated hex strings)
62    /// Can also be set via TENUO_TRUSTED_KEYS env var
63    #[arg(long, env = "TENUO_TRUSTED_KEYS")]
64    trusted_keys: Option<String>,
65
66    /// Path to revocation list file (JSON array of warrant IDs)
67    /// Can also be set via TENUO_REVOCATION_LIST env var
68    #[arg(long, env = "TENUO_REVOCATION_LIST")]
69    revocation_list: Option<PathBuf>,
70
71    #[command(subcommand)]
72    command: Commands,
73}
74
75#[derive(Subcommand)]
76enum Commands {
77    /// Verify and authorize a single warrant (for scripting)
78    Verify {
79        /// Warrant (base64, or - for stdin)
80        #[arg(short, long)]
81        warrant: Option<String>,
82
83        /// Tool name to authorize
84        #[arg(short, long)]
85        tool: String,
86
87        /// Arguments in key=value format
88        #[arg(short, long)]
89        arg: Vec<String>,
90
91        /// Output format: exit-code, json, or quiet
92        #[arg(short, long, default_value = "exit-code")]
93        output: String,
94    },
95
96    /// Check if a warrant is valid (no authorization, just verification)
97    Check {
98        /// Warrant (base64, or - for stdin)
99        #[arg(short, long)]
100        warrant: Option<String>,
101    },
102
103    /// Print authorizer info
104    Info,
105}
106
107fn main() -> Result<(), Box<dyn std::error::Error>> {
108    let cli = Cli::parse();
109
110    // Build authorizer from trusted keys and revocation list
111    let authorizer = build_authorizer(&cli.trusted_keys, &cli.revocation_list)?;
112
113    match cli.command {
114        Commands::Verify {
115            warrant,
116            tool,
117            arg,
118            output,
119        } => {
120            // Read warrant
121            let warrant_str = read_warrant(warrant)?;
122            let w = wire::decode_base64(&warrant_str)?;
123
124            // Parse arguments
125            let mut args = HashMap::new();
126            for a in arg {
127                let parts: Vec<&str> = a.splitn(2, '=').collect();
128                if parts.len() != 2 {
129                    return Err(format!("Invalid argument format: {}", a).into());
130                }
131                args.insert(
132                    parts[0].to_string(),
133                    ConstraintValue::String(parts[1].to_string()),
134                );
135            }
136
137            // Check authorization (no approvals for CLI mode)
138            let result = authorizer.check(&w, &tool, &args, None, &[]);
139
140            match output.as_str() {
141                "exit-code" => {
142                    if result.is_ok() {
143                        std::process::exit(0);
144                    } else {
145                        eprintln!("Authorization failed: {}", result.unwrap_err());
146                        std::process::exit(1);
147                    }
148                }
149                "json" => {
150                    let json = serde_json::json!({
151                        "authorized": result.is_ok(),
152                        "error": result.err().map(|e| e.to_string()),
153                        "warrant_id": w.id().as_str(),
154                        "tool": tool,
155                    });
156                    println!("{}", serde_json::to_string_pretty(&json)?);
157                }
158                "quiet" => {
159                    if result.is_err() {
160                        std::process::exit(1);
161                    }
162                }
163                _ => return Err(format!("Unknown output format: {}", output).into()),
164            }
165        }
166
167        Commands::Check { warrant } => {
168            let warrant_str = read_warrant(warrant)?;
169            let w = wire::decode_base64(&warrant_str)?;
170
171            // Just verify, don't authorize
172            match w.verify(w.issuer()) {
173                Ok(()) => {
174                    println!("✓ Warrant signature is valid");
175                    println!("  ID: {}", w.id());
176                    println!("  Tool: {}", w.tool());
177                    println!("  Expires: {}", w.expires_at());
178                    if w.is_expired() {
179                        println!("  ⚠ WARNING: Warrant has EXPIRED");
180                        std::process::exit(2);
181                    }
182                }
183                Err(e) => {
184                    eprintln!("✗ Warrant invalid: {}", e);
185                    std::process::exit(1);
186                }
187            }
188        }
189
190        Commands::Info => {
191            println!("Tenuo Authorizer v{}", env!("CARGO_PKG_VERSION"));
192            println!();
193            if let Some(keys) = &cli.trusted_keys {
194                let count = keys.split(',').filter(|s| !s.is_empty()).count();
195                println!("Trusted keys: {} configured", count);
196            } else {
197                println!("Trusted keys: None (will accept any issuer for delegated warrants)");
198            }
199            if let Some(path) = &cli.revocation_list {
200                println!("Revocation list: {}", path.display());
201            } else {
202                println!("Revocation list: None");
203            }
204        }
205    }
206
207    Ok(())
208}

210fn build_authorizer(
211    trusted_keys: &Option<String>,
212    revocation_path: &Option<PathBuf>,
213) -> Result<Authorizer, Box<dyn std::error::Error>> {
214    // Start with a dummy authorizer if no keys provided
215    // This still validates signatures, just doesn't check the issuer
216    let first_key = if let Some(keys) = trusted_keys {
217        let first = keys.split(',').next().unwrap_or("");
218        if first.is_empty() {
219            return Err("TENUO_TRUSTED_KEYS is empty".into());
220        }
221        let bytes = hex::decode(first)?;
222        let arr: [u8; 32] = bytes.try_into().map_err(|_| "invalid key length")?;
223        PublicKey::from_bytes(&arr)?
224    } else {
225        // For development: create a dummy key
226        // In production, TENUO_TRUSTED_KEYS should always be set
227        eprintln!("WARNING: No trusted keys configured. Set TENUO_TRUSTED_KEYS for production.");
228        let dummy = [0u8; 32];
229        PublicKey::from_bytes(&dummy).unwrap_or_else(|_| {
230            // Generate a valid but useless key
231            tenuo_core::Keypair::generate().public_key()
232        })
233    };
234
235    let mut authorizer = Authorizer::new(first_key);
236
237    // Add remaining keys
238    if let Some(keys) = trusted_keys {
239        for key_hex in keys.split(',').skip(1) {
240            if !key_hex.is_empty() {
241                let bytes = hex::decode(key_hex)?;
242                let arr: [u8; 32] = bytes.try_into().map_err(|_| "invalid key length")?;
243                authorizer.add_trusted_key(PublicKey::from_bytes(&arr)?);
244            }
245        }
246    }
247
248    // Load revocation list if provided
249    if let Some(path) = revocation_path {
250        let revocation_list = load_revocation_list(path)?;
251        authorizer.set_revocation_list(revocation_list);
252        eprintln!("Loaded revocation list from: {}", path.display());
253    }
254
255    Ok(authorizer)
256}

258fn load_revocation_list(path: &PathBuf) -> Result<RevocationList, Box<dyn std::error::Error>> {
259    let content = std::fs::read_to_string(path)?;
260    
261    // Parse as JSON array of warrant IDs
262    let ids: Vec<String> = serde_json::from_str(&content)?;
263    
264    let mut list = RevocationList::new();
265    for id in ids {
266        list.revoke(&id);
267    }
268    
269    Ok(list)
270}

272fn read_warrant(warrant: Option<String>) -> Result<String, Box<dyn std::error::Error>> {
273    match warrant {
274        Some(w) if w == "-" => {
275            let mut buf = String::new();
276            io::stdin().read_to_string(&mut buf)?;
277            Ok(buf.trim().to_string())
278        }
279        Some(w) => Ok(w),
280        None => {
281            // Try stdin
282            let mut buf = String::new();
283            io::stdin().read_to_string(&mut buf)?;
284            Ok(buf.trim().to_string())
285        }
286    }
287}

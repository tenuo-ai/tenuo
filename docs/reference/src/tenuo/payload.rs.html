10/// The payload of a warrant (unsigned).
11///
12/// Refactored to separate file to avoid macro scope issues.
13#[derive(Debug, Clone)]
14pub struct WarrantPayload {
15    /// Schema version for this warrant.
16    pub version: u8,
17    /// Type of warrant: ISSUER or EXECUTION.
18    pub warrant_type: WarrantType,
19    /// Unique identifier for this warrant.
20    pub id: WarrantId,
21    /// Tools: Map of tool_name -> constraint_set.
22    ///
23    /// This replaces the separate `tools` and `constraints` fields, allowing
24    /// for granular, per-tool constraints.
25    pub tools: BTreeMap<String, ConstraintSet>,
26    /// Authorized holder's public key.
27    pub holder: PublicKey,
28    /// Issuer's public key.
29    pub issuer: PublicKey,
30    /// Unix timestamp (seconds) when issued.
31    pub issued_at: u64,
32    /// Unix timestamp (seconds) when expires.
33    pub expires_at: u64,
34    /// Maximum delegation depth (policy limit, decreases on attenuation).
35    pub max_depth: u8,
36    /// Current delegation depth (0 for root, increments on attenuation).
37    pub depth: u32,
38    /// Hash of the parent warrant's payload (for tracking/auditing).
39    /// Content-addressed linkage to parent.
40    pub parent_hash: Option<[u8; 32]>,
41    /// Extension data (arbitrary bytes).
42    pub extensions: BTreeMap<String, Vec<u8>>,
43
44    // Issuer Warrant Fields
45    pub issuable_tools: Option<Vec<String>>,
46    pub max_issue_depth: Option<u32>,
47    pub constraint_bounds: Option<ConstraintSet>,
48
49    // Common Fields
50    pub clearance: Option<Clearance>,
51    pub session_id: Option<String>,
52    pub agent_id: Option<String>,
53    pub required_approvers: Option<Vec<PublicKey>>,
54    pub min_approvals: Option<u32>,
55}

57// Integer key mapping (wire format):
58// 0: version
59// 1: id
60// 2: warrant_type
61// 3: tools
62// 4: holder
63// 5: issuer
64// 6: issued_at
65// 7: expires_at
66// 8: max_depth
67// 9: parent_hash
68// 10: extensions
69// 11: issuable_tools
70// 12: (removed - was trust_ceiling)
71// 13: max_issue_depth
72// 14: constraint_bounds
73// 15: required_approvers
74// 16: min_approvals
75// 17: clearance
76// 18: depth
77// Metadata fields not in authz-critical path (session_id, agent_id)
78// are serialized into extensions with reserved keys:
79const EXT_KEY_SESSION_ID: &str = "tenuo.session_id";
80const EXT_KEY_AGENT_ID: &str = "tenuo.agent_id";

82impl Serialize for WarrantPayload {
83    fn serialize<S>(&self, serializer: S) -> std::result::Result<S::Ok, S::Error>
84    where
85        S: Serializer,
86    {
87        // Prepare extensions with metadata folded in
88        let mut extensions = self.extensions.clone();
89        if let Some(session) = &self.session_id {
90            extensions.insert(EXT_KEY_SESSION_ID.to_string(), session.as_bytes().to_vec());
91        }
92        if let Some(agent) = &self.agent_id {
93            extensions.insert(EXT_KEY_AGENT_ID.to_string(), agent.as_bytes().to_vec());
94        }
95
96        // Count fields to serialize
97        let mut entries = 12; // required fields 0-10 + depth at 18
98        if self.parent_hash.is_none() {
99            entries -= 1;
100        }
101        if extensions.is_empty() {
102            entries -= 1;
103        }
104        if self.issuable_tools.is_some() {
105            entries += 1;
106        }
107        if self.max_issue_depth.is_some() {
108            entries += 1;
109        }
110        if self.constraint_bounds.is_some() {
111            entries += 1;
112        }
113        if self.required_approvers.is_some() {
114            entries += 1;
115        }
116        if self.min_approvals.is_some() {
117            entries += 1;
118        }
119        if self.clearance.is_some() {
120            entries += 1;
121        }
122
123        let mut map = serializer.serialize_map(Some(entries))?;
124        map.serialize_entry(&0u8, &self.version)?;
125        map.serialize_entry(&1u8, &self.id)?;
126        map.serialize_entry(&2u8, &self.warrant_type)?;
127        map.serialize_entry(&3u8, &self.tools)?;
128        map.serialize_entry(&4u8, &self.holder)?;
129        map.serialize_entry(&5u8, &self.issuer)?;
130        map.serialize_entry(&6u8, &self.issued_at)?;
131        map.serialize_entry(&7u8, &self.expires_at)?;
132        map.serialize_entry(&8u8, &self.max_depth)?;
133        if let Some(parent_hash) = &self.parent_hash {
134            map.serialize_entry(&9u8, parent_hash)?;
135        }
136        if !extensions.is_empty() {
137            map.serialize_entry(&10u8, &extensions)?;
138        }
139        if let Some(issuable) = &self.issuable_tools {
140            map.serialize_entry(&11u8, issuable)?;
141        }
142        if let Some(max_issue) = &self.max_issue_depth {
143            map.serialize_entry(&13u8, max_issue)?;
144        }
145        if let Some(bounds) = &self.constraint_bounds {
146            map.serialize_entry(&14u8, bounds)?;
147        }
148        if let Some(req) = &self.required_approvers {
149            map.serialize_entry(&15u8, req)?;
150        }
151        if let Some(min) = &self.min_approvals {
152            map.serialize_entry(&16u8, min)?;
153        }
154        if let Some(c) = &self.clearance {
155            map.serialize_entry(&17u8, c)?;
156        }
157        map.serialize_entry(&18u8, &self.depth)?;
158        map.end()
159    }
160}

162impl<'de> Deserialize<'de> for WarrantPayload {
163    fn deserialize<D>(deserializer: D) -> std::result::Result<Self, D::Error>
164    where
165        D: Deserializer<'de>,
166    {
167        struct WPVisitor;
168
169        impl<'de> Visitor<'de> for WPVisitor {
170            type Value = WarrantPayload;
171
172            fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
173                formatter.write_str("CBOR map of warrant payload with integer keys")
174            }
175
176            fn visit_map<A>(self, mut map: A) -> std::result::Result<Self::Value, A::Error>
177            where
178                A: MapAccess<'de>,
179            {
180                let mut seen = HashSet::new();
181
182                let mut version = None;
183                let mut id = None;
184                let mut warrant_type = None;
185                let mut tools = None;
186                let mut holder = None;
187                let mut issuer = None;
188                let mut issued_at = None;
189                let mut expires_at = None;
190                let mut max_depth = None;
191                let mut parent_hash = None;
192                let mut extensions: BTreeMap<String, Vec<u8>> = BTreeMap::new();
193                let mut issuable_tools = None;
194                let mut max_issue_depth = None;
195                let mut constraint_bounds = None;
196                let mut required_approvers = None;
197                let mut min_approvals = None;
198                let mut clearance = None;
199                let mut depth: Option<u32> = None;
200
201                while let Some(key) = map.next_key::<u8>()? {
202                    if !seen.insert(key) {
203                        return Err(A::Error::custom(format!("duplicate key {}", key)));
204                    }
205                    match key {
206                        0 => version = Some(map.next_value()?),
207                        1 => id = Some(map.next_value()?),
208                        2 => warrant_type = Some(map.next_value()?),
209                        3 => tools = Some(map.next_value()?),
210                        4 => holder = Some(map.next_value()?),
211                        5 => issuer = Some(map.next_value()?),
212                        6 => issued_at = Some(map.next_value()?),
213                        7 => expires_at = Some(map.next_value()?),
214                        8 => max_depth = Some(map.next_value()?),
215                        9 => parent_hash = map.next_value()?,
216                        10 => extensions = map.next_value()?,
217                        11 => issuable_tools = map.next_value()?,
218                        // 12 was trust_ceiling, removed in alpha.7
219                        13 => max_issue_depth = map.next_value()?,
220                        14 => constraint_bounds = map.next_value()?,
221                        15 => required_approvers = map.next_value()?,
222                        16 => min_approvals = map.next_value()?,
223                        17 => clearance = map.next_value()?,
224                        18 => depth = Some(map.next_value()?),
225                        _ => {
226                            // Spec requires FAIL CLOSED on unknown keys
227                            return Err(A::Error::custom(format!(
228                                "unknown payload field key {}",
229                                key
230                            )));
231                        }
232                    }
233                }
234
235                let version = version.ok_or_else(|| A::Error::custom("missing version"))?;
236                let id = id.ok_or_else(|| A::Error::custom("missing id"))?;
237                let warrant_type =
238                    warrant_type.ok_or_else(|| A::Error::custom("missing warrant_type"))?;
239                let tools = tools.ok_or_else(|| A::Error::custom("missing tools"))?;
240                let holder = holder.ok_or_else(|| A::Error::custom("missing holder"))?;
241                let issuer = issuer.ok_or_else(|| A::Error::custom("missing issuer"))?;
242                let issued_at = issued_at.ok_or_else(|| A::Error::custom("missing issued_at"))?;
243                let expires_at =
244                    expires_at.ok_or_else(|| A::Error::custom("missing expires_at"))?;
245                let max_depth = max_depth.ok_or_else(|| A::Error::custom("missing max_depth"))?;
246
247                // Extract metadata from extensions
248                let mut session_id = None;
249                let mut agent_id = None;
250                if let Some(bytes) = extensions.remove(EXT_KEY_SESSION_ID) {
251                    session_id = String::from_utf8(bytes)
252                        .map(Some)
253                        .map_err(|e| A::Error::custom(format!("invalid session_id utf8: {}", e)))?;
254                }
255                if let Some(bytes) = extensions.remove(EXT_KEY_AGENT_ID) {
256                    agent_id = String::from_utf8(bytes)
257                        .map(Some)
258                        .map_err(|e| A::Error::custom(format!("invalid agent_id utf8: {}", e)))?;
259                }
260
261                Ok(WarrantPayload {
262                    version,
263                    warrant_type,
264                    id,
265                    tools,
266                    holder,
267                    issuer,
268                    issued_at,
269                    expires_at,
270                    max_depth,
271                    depth: depth.unwrap_or(0), // 0 for root warrants
272                    parent_hash,
273                    extensions,
274                    issuable_tools,
275                    max_issue_depth,
276                    constraint_bounds,
277                    clearance,
278                    session_id,
279                    agent_id,
280                    required_approvers,
281                    min_approvals,
282                })
283            }
284        }
285
286        deserializer.deserialize_map(WPVisitor)
287    }

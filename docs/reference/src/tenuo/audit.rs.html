1//! Audit Logging infrastructure.
2//!
3//! Provides traits and implementations for logging security-critical events.
4
5use crate::approval::AuditEvent;
6use std::sync::{Arc, RwLock};
7
8/// Trait for audit loggers.
9pub trait AuditLogger: Send + Sync + std::fmt::Debug {
10    /// Log an audit event.
11    fn log(&self, event: AuditEvent);
12}
13
14/// A logger that writes events to stdout as JSON lines.
15///
16/// This is suitable for containerized environments (Kubernetes, Docker) where
17/// logs are scraped by an external agent (Fluentd, Datadog, etc.).
18#[derive(Debug, Default)]
19pub struct StdoutLogger;
20
21impl StdoutLogger {
22    pub fn new() -> Self {
23        Self
24    }
25}
26
27impl AuditLogger for StdoutLogger {
28    fn log(&self, event: AuditEvent) {
29        if let Ok(json) = serde_json::to_string(&event) {
30            println!("{}", json);
31        } else {
32            eprintln!("Failed to serialize audit event: {:?}", event);
33        }
34    }
35}
36
37/// A logger that does nothing (for testing or when auditing is disabled).
38#[derive(Debug, Default)]
39pub struct NoOpLogger;
40
41impl AuditLogger for NoOpLogger {
42    fn log(&self, _event: AuditEvent) {}
43}
44
45/// Global audit logger instance.
46///
47/// We use a global logger to avoid threading it through every function call.
48/// It is initialized to NoOpLogger by default.
49static GLOBAL_LOGGER: RwLock<Option<Arc<dyn AuditLogger>>> = RwLock::new(None);
50
51/// Set the global audit logger.
52pub fn set_global_logger(logger: Arc<dyn AuditLogger>) {
53    let mut lock = GLOBAL_LOGGER.write().unwrap();
54    *lock = Some(logger);
55}
56
57/// Log an event using the global logger.
58pub fn log_event(event: AuditEvent) {
59    if let Ok(lock) = GLOBAL_LOGGER.read() {
60        if let Some(logger) = lock.as_ref() {
61            logger.log(event);
62        }
63    }
64}

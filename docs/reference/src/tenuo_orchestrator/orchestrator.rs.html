1//! Orchestrator Agent Demo
2//!
3//! This demonstrates how an orchestrator agent:
4//! 1. Receives a broad warrant from the Control Plane (with max_depth policy)
5//! 2. Attenuates it to create a narrower warrant for a worker
6//! 3. Delegates the attenuated warrant via a chain with session binding
7//! 4. Shows how depth limits prevent unbounded delegation
8//! 5. **Multi-sig approval**: Requires human approval for sensitive actions
9//! 6. **Notary Registry**: Maps enterprise identities to cryptographic keys
10
11use tenuo_core::{Keypair, Pattern, Exact, Range, Warrant, wire, OneOf, Wildcard};
12use std::time::Duration;
13use std::env;
14use uuid::Uuid;
15
16fn main() -> Result<(), Box<dyn std::error::Error>> {
17    println!("\n╔══════════════════════════════════════════════════════════════════╗");
18    println!("║                    ORCHESTRATOR AGENT DEMO                       ║");
19    println!("║          Demonstrating Capability Delegation in Tenuo            ║");
20    println!("╚══════════════════════════════════════════════════════════════════╝\n");
21
22    // Generate a session ID for this workflow (for traceability)
23    let session_id = format!("sess_{}", Uuid::now_v7().simple());
24    println!("  Session ID: {}\n", session_id);
25
26    // =========================================================================
27    // Step 1: Simulate receiving a root warrant from the Control Plane
28    // =========================================================================
29    println!("┌─────────────────────────────────────────────────────────────────┐");
30    println!("│ STEP 1: Receiving Root Warrant from Control Plane               │");
31    println!("└─────────────────────────────────────────────────────────────────┘");
32
33    let cp_key = env::var("TENUO_SECRET_KEY")
34        .expect("TENUO_SECRET_KEY must be set (hex-encoded 32-byte key)");
35    let cp_key_bytes: [u8; 32] = hex::decode(&cp_key)?
36        .try_into()
37        .map_err(|_| "Secret key must be 32 bytes")?;
38    
39    let cp_keypair = Keypair::from_bytes(&cp_key_bytes);
40
41    println!("  Control Plane Public Key: {}", hex::encode(cp_keypair.public_key().to_bytes()));
42
43    // The Control Plane issues a broad warrant with POLICY LIMITS
44    // action uses Wildcard - can be attenuated to any constraint type
45    let root_warrant = Warrant::builder()
46        .tool("manage_infrastructure")
47        .constraint("cluster", Pattern::new("staging-*")?)
48        .constraint("action", Wildcard::new())  // Wildcard: can narrow to anything
49        .constraint("budget", Range::max(10000.0))
50        .ttl(Duration::from_secs(3600))
51        .max_depth(3)                          // POLICY: Only 3 levels of delegation allowed!
52        .session_id(session_id.clone())        // Session binding for traceability
53        .agent_id("orchestrator-main")         // Agent ID for audit
54        .build(&cp_keypair)?;
55
56    println!("\n  ✓ Root Warrant Received:");
57    println!("    • ID:          {}", root_warrant.id());
58    println!("    • Tool:        {}", root_warrant.tool());
59    println!("    • Depth:       {} (root)", root_warrant.depth());
60    println!("    • Max Depth:   {} (policy limit)", root_warrant.effective_max_depth());
61    println!("    • Session:     {}", root_warrant.session_id().unwrap_or("-"));
62    println!("    • Expires:     {}", root_warrant.expires_at());
63    println!("    • Constraints:");
64    println!("      - cluster:   staging-*");
65    println!("      - action:    * (Wildcard)");
66    println!("      - budget:    ≤ $10,000");
67
68    // =========================================================================
69    // Step 2: Orchestrator creates its own identity
70    // =========================================================================
71    println!("\n┌─────────────────────────────────────────────────────────────────┐");
72    println!("│ STEP 2: Orchestrator Establishes Identity                       │");
73    println!("└─────────────────────────────────────────────────────────────────┘");
74
75    let orchestrator_keypair = Keypair::generate();
76    println!("  Generated Orchestrator Keypair");
77    println!("    Public Key: {}", hex::encode(orchestrator_keypair.public_key().to_bytes()));
78
79    // Generate worker keypair (for PoP)
80    let worker_keypair = Keypair::generate();
81    println!("  Generated Worker Keypair (for PoP)");
82    println!("    Public Key: {}", hex::encode(worker_keypair.public_key().to_bytes()));
83
84    // Save worker key for the worker agent
85    let worker_key_path = env::var("TENUO_WORKER_KEY_OUTPUT")
86        .unwrap_or_else(|_| "/data/worker.key".to_string());
87    std::fs::write(&worker_key_path, hex::encode(worker_keypair.secret_key_bytes()))?;
88    println!("    Saved to: {}", worker_key_path);
89
90    // =========================================================================
91    // Step 3: Attenuate warrant for the Worker
92    // =========================================================================
93    println!("\n┌─────────────────────────────────────────────────────────────────┐");
94    println!("│ STEP 3: Attenuating Warrant for Worker                          │");
95    println!("└─────────────────────────────────────────────────────────────────┘");
96
97    println!("  Narrowing capabilities:");
98    println!("    • cluster: staging-*  → staging-web (exact)");
99    println!("    • action:  * (any)    → [upgrade, restart] (OneOf)");
100    println!("    • budget:  ≤$10,000   → ≤$1,000 (reduced)");
101    println!("    • TTL:     1 hour     → 10 minutes (shortened)");
102
103    let worker_warrant = root_warrant.attenuate()
104        .constraint("cluster", Exact::new("staging-web"))
105        .constraint("action", OneOf::new(vec!["upgrade", "restart"]))
106        .constraint("budget", Range::max(1000.0))
107        .ttl(Duration::from_secs(600)) // 10 minutes
108        .authorized_holder(worker_keypair.public_key()) // PoP
109        .agent_id("worker-agent-01") // Traceability
110        // Session ID is inherited from parent automatically
111        .build(&orchestrator_keypair)?;
112
113    println!("\n  ✓ Worker Warrant Created:");
114    println!("    • ID:          {}", worker_warrant.id());
115    println!("    • Parent ID:   {}", worker_warrant.parent_id().unwrap());
116    println!("    • Depth:       {} / {} (delegated)", worker_warrant.depth(), worker_warrant.effective_max_depth());
117    println!("    • Session:     {} (inherited)", worker_warrant.session_id().unwrap_or("-"));
118    println!("    • Expires:     {}", worker_warrant.expires_at());
119    println!("    • Holder:      {} (PoP required)", hex::encode(&worker_keypair.public_key().to_bytes()[..8]));
120    println!("    • Signed by:   Orchestrator");
121
122    // =========================================================================
123    // Step 4: Multi-Sig Warrant (Sensitive Actions with Human Approval)
124    // =========================================================================
125    println!("\n┌─────────────────────────────────────────────────────────────────┐");
126    println!("│ STEP 4: Creating Multi-Sig Warrant (Human Approval Required)    │");
127    println!("└─────────────────────────────────────────────────────────────────┘");
128
129    println!("  For sensitive actions (delete, scale-down), we require human approval.");
130    println!("  This is enforced via multi-sig: the warrant lists required approvers.\n");
131
132    // Create an "approver" keypair (simulates an admin mapped via Notary Registry)
133    let admin_keypair = Keypair::generate();
134    println!("  Generated Admin Keypair (simulating Notary-bound identity):");
135    println!("    • Public Key: {}", hex::encode(&admin_keypair.public_key().to_bytes()[..16]));
136    println!("    • External ID: arn:aws:iam::123456789:user/admin (simulated)\n");
137
138    // Save admin key for the worker demo
139    let admin_key_path = env::var("TENUO_ADMIN_KEY_OUTPUT")
140        .unwrap_or_else(|_| "/data/admin.key".to_string());
141    std::fs::write(&admin_key_path, hex::encode(admin_keypair.secret_key_bytes()))?;
142    println!("    Saved admin key to: {}", admin_key_path);
143
144    // Create a warrant for sensitive operations that REQUIRES multi-sig approval
145    let sensitive_warrant = root_warrant.attenuate()
146        .constraint("cluster", Exact::new("staging-web"))
147        .constraint("action", OneOf::new(vec!["delete", "scale-down"]))  // Dangerous actions
148        .constraint("budget", Range::max(500.0))
149        .ttl(Duration::from_secs(300)) // 5 minutes (short for sensitive ops)
150        .authorized_holder(worker_keypair.public_key())
151        .agent_id("worker-agent-01-sensitive")
152        // MULTI-SIG: Require 1-of-1 approval from the admin
153        .add_approvers(vec![admin_keypair.public_key()])
154        .raise_min_approvals(1)
155        .build(&orchestrator_keypair)?;
156
157    println!("\n  ✓ Multi-Sig Warrant Created:");
158    println!("    • ID:          {}", sensitive_warrant.id());
159    println!("    • Actions:     [delete, scale-down] (dangerous!)");
160    println!("    • Budget:      ≤$500");
161    println!("    • Approvers:   1 (admin required)");
162    println!("    • Threshold:   1-of-1");
163    println!("    • TTL:         5 minutes (short-lived for safety)");
164
165    // =========================================================================
166    // Step 5: Create and output the delegation chain
167    // =========================================================================
168    println!("\n┌─────────────────────────────────────────────────────────────────┐");
169    println!("│ STEP 5: Creating Delegation Chain                               │");
170    println!("└─────────────────────────────────────────────────────────────────┘");
171
172    let chain = vec![root_warrant.clone(), worker_warrant.clone()];
173    let sensitive_chain = vec![root_warrant.clone(), sensitive_warrant.clone()];
174    
175    println!("  Chain Structure:");
176    println!("    ┌─────────────────────────────────────────────────────────────┐");
177    println!("    │  [0] Root Warrant (Control Plane)                           │");
178    println!("    │      └── cluster: staging-*, action: *, budget: ≤$10k       │");
179    println!("    │              │                                              │");
180    println!("    │              ▼  (attenuation)                               │");
181    println!("    │  [1] Worker Warrant (Orchestrator)                          │");
182    println!("    │      └── cluster: staging-web, action: upgrade|restart,     │");
183    println!("    │          budget: ≤$1k                                       │");
184    println!("    └─────────────────────────────────────────────────────────────┘");
185
186    // Serialize the chains for the worker
187    let chain_json = serde_json::to_string_pretty(&chain)?;
188    let sensitive_chain_json = serde_json::to_string_pretty(&sensitive_chain)?;
189    
190    let output_path = env::var("TENUO_CHAIN_OUTPUT")
191        .unwrap_or_else(|_| "/data/chain.json".to_string());
192    std::fs::write(&output_path, &chain_json)?;
193    println!("\n  ✓ Standard chain written to: {}", output_path);
194
195    let sensitive_output_path = output_path.replace(".json", "_sensitive.json");
196    std::fs::write(&sensitive_output_path, &sensitive_chain_json)?;
197    println!("  ✓ Sensitive chain written to: {}", sensitive_output_path);
198
199    // Also output the warrant in wire format (base64)
200    let wire_format = wire::encode_base64(&worker_warrant)?;
201    println!("\n  Worker Warrant (wire format, {} bytes):", wire_format.len());
202    println!("    {}", &wire_format[..80.min(wire_format.len())]);
203    if wire_format.len() > 80 {
204        println!("    ... ({} more bytes)", wire_format.len() - 80);
205    }
206
207    // =========================================================================
208    // Summary
209    // =========================================================================
210    println!("\n╔══════════════════════════════════════════════════════════════════╗");
211    println!("║                     ORCHESTRATOR COMPLETE                        ║");
212    println!("╠══════════════════════════════════════════════════════════════════╣");
213    println!("║  Features Demonstrated:                                          ║");
214    println!("║  1. max_depth(3)       - Policy limits delegation depth          ║");
215    println!("║  2. session_id         - Warrants share same session             ║");
216    println!("║  3. agent_id           - Agents are identifiable for audit       ║");
217    println!("║  4. holder_key         - Worker must prove possession (PoP)      ║");
218    println!("║  5. MULTI-SIG          - 1-of-1 admin approval for delete!       ║");
219    println!("║                                                                  ║");
220    println!("║  The Worker will demonstrate:                                    ║");
221    println!("║  • Standard actions (upgrade/restart) → no approval needed       ║");
222    println!("║  • Sensitive actions (delete) → BLOCKED without approval         ║");
223    println!("║  • Sensitive actions (delete) → ALLOWED with admin approval      ║");
224    println!("╚══════════════════════════════════════════════════════════════════╝\n");
225    
226    Ok(())
227}

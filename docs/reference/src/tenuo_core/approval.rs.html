1//! Approval types for human-in-the-loop and multi-sig workflows.
2//!
3//! Tenuo treats Identity Providers as "Notaries" - they map enterprise
4//! identities (AWS IAM ARNs, Okta users, YubiKey certificates) to Ed25519
5//! public keys. Tenuo only cares about the cryptographic signature.
6//!
7//! ## Philosophy
8//!
9//! ```text
10//! ┌─────────────────────────────────────────────────────────────────┐
11//! │  IDENTITY PROVIDER (Notary)          TENUO CORE (The Math)     │
12//! │  ─────────────────────────           ──────────────────────    │
13//! │                                                                 │
14//! │  "arn:aws:iam::123:user/admin"  →    PublicKey [32 bytes]      │
15//! │  "okta:user:alice@corp.com"     →    PublicKey [32 bytes]      │
16//! │  "yubikey:serial:12345678"      →    PublicKey [32 bytes]      │
17//! │                                                                 │
18//! │  Provider handles:                   Tenuo verifies:           │
19//! │  • Identity verification             • Signature math          │
20//! │  • Key derivation/storage            • Multi-sig counting      │
21//! │  • Key rotation                      • Offline verification    │
22//! └─────────────────────────────────────────────────────────────────┘
23//! ```
24//!
25//! ## Notary Registry
26//!
27//! The `NotaryRegistry` manages the lifecycle of identity-to-key bindings:
28//!
29//! ```rust,ignore
30//! let mut registry = NotaryRegistry::new();
31//!
32//! // Register an AWS IAM user's key
33//! registry.register_key(KeyBinding {
34//!     external_id: "arn:aws:iam::123:user/admin".into(),
35//!     provider: "aws-iam".into(),
36//!     public_key: admin_keypair.public_key(),
37//!     registered_by: "bootstrap".into(),
38//!     // ...
39//! })?;
40//!
41//! // All operations emit audit events
42//! for event in registry.drain_events() {
43//!     audit_log.record(event);
44//! }
45//! ```
46//!
47//! ## Implementation Status
48//!
49//! - [x] Approval struct (data model)
50//! - [x] NotaryRegistry (key lifecycle management)
51//! - [x] KeyBinding (identity → key mapping)
52//! - [x] AuditEvent (key lifecycle auditing)
53//! - [ ] Multi-sig verification in Authorizer
54//! - [ ] Python SDK: ApprovalProvider ABC
55//! - [ ] Provider implementations (AWS IAM, Okta, YubiKey)

62/// A cryptographically signed approval from a human or external system.
63///
64/// The approval is bound to a specific request (via `request_hash`) and
65/// signed by an approver's keypair. The approver's identity is tracked
66/// via `external_id` for audit purposes.
67#[derive(Debug, Clone, Serialize, Deserialize)]
68pub struct Approval {
69    /// Hash of what was approved: H(warrant_id || tool || sorted(args))
70    pub request_hash: [u8; 32],
71
72    /// The approver's public key
73    pub approver_key: PublicKey,
74
75    /// External identity reference (e.g., "arn:aws:iam::123:user/admin")
76    pub external_id: String,
77
78    /// Provider name (e.g., "aws-iam", "okta", "yubikey")
79    pub provider: String,
80
81    /// When approved (UTC)
82    pub approved_at: DateTime<Utc>,
83
84    /// When this approval expires (UTC)
85    pub expires_at: DateTime<Utc>,
86
87    /// Optional: human-readable reason/justification
88    #[serde(skip_serializing_if = "Option::is_none")]
89    pub reason: Option<String>,
90
91    /// Signature over the approval payload
92    pub signature: Signature,
93}
94
95impl Approval {
96    /// Verify the approval signature and check expiration.
97    pub fn verify(&self) -> Result<()> {
98        // Check expiration
99        if Utc::now() > self.expires_at {
100            return Err(Error::ApprovalExpired {
101                approved_at: self.approved_at,
102                expired_at: self.expires_at,
103            });
104        }
105
106        // Verify signature
107        let payload = self.signable_bytes();
108        self.approver_key.verify(&payload, &self.signature)
109    }
110
111    /// Check if this approval matches a given request.
112    pub fn matches_request(&self, request_hash: &[u8; 32]) -> bool {
113        &self.request_hash == request_hash
114    }
115
116    /// Get the bytes that were signed.
117    fn signable_bytes(&self) -> Vec<u8> {
118        // Canonical format: request_hash || external_id || approved_at || expires_at
119        let mut bytes = Vec::new();
120        bytes.extend_from_slice(&self.request_hash);
121        bytes.extend_from_slice(self.external_id.as_bytes());
122        bytes.extend_from_slice(&self.approved_at.timestamp().to_le_bytes());
123        bytes.extend_from_slice(&self.expires_at.timestamp().to_le_bytes());
124        bytes
125    }
126}

128/// Compute a request hash for approval binding.
129///
130/// This ensures an approval is bound to a specific (warrant, tool, args, holder) tuple.
131/// Including the holder prevents approval theft: even if an attacker intercepts an
132/// approval, they can't use it because the hash won't match their holder key.
133pub fn compute_request_hash(
134    warrant_id: &str,
135    tool: &str,
136    args: &std::collections::HashMap<String, crate::constraints::ConstraintValue>,
137    authorized_holder: Option<&crate::crypto::PublicKey>,
138) -> [u8; 32] {
139    use sha2::{Digest, Sha256};
140    use std::collections::BTreeMap;
141
142    let mut hasher = Sha256::new();
143    hasher.update(warrant_id.as_bytes());
144    hasher.update(b"|");
145    hasher.update(tool.as_bytes());
146    hasher.update(b"|");
147
148    // Sort args for deterministic hashing
149    let sorted: BTreeMap<_, _> = args.iter().collect();
150    if let Ok(json) = serde_json::to_vec(&sorted) {
151        hasher.update(&json);
152    }
153
154    // Bind to authorized holder (prevents approval theft)
155    hasher.update(b"|");
156    if let Some(holder) = authorized_holder {
157        hasher.update(holder.to_bytes());
158    }
159
160    hasher.finalize().into()
161}

167/// A binding between an external identity and a Tenuo public key.
168///
169/// This is the core data structure for the Notary Registry, mapping
170/// enterprise identities (IAM ARNs, Okta users, etc.) to Ed25519 keys.
171#[derive(Debug, Clone, Serialize, Deserialize)]
172pub struct KeyBinding {
173    /// Unique ID for this binding (e.g., "kb_<uuid>")
174    pub id: String,
175
176    /// External identity (e.g., "arn:aws:iam::123:user/admin")
177    pub external_id: String,
178
179    /// Provider name (e.g., "aws-iam", "okta", "yubikey")
180    pub provider: String,
181
182    /// The bound public key
183    pub public_key: PublicKey,
184
185    /// Human-readable display name
186    #[serde(skip_serializing_if = "Option::is_none")]
187    pub display_name: Option<String>,
188
189    /// Who registered this binding
190    pub registered_by: String,
191
192    /// When this binding was created
193    pub registered_at: DateTime<Utc>,
194
195    /// When this binding expires (None = never)
196    #[serde(skip_serializing_if = "Option::is_none")]
197    pub expires_at: Option<DateTime<Utc>>,
198
199    /// Whether this binding is currently active
200    pub active: bool,
201
202    /// Optional metadata (tags, permissions, etc.)
203    #[serde(skip_serializing_if = "Option::is_none")]
204    pub metadata: Option<std::collections::BTreeMap<String, String>>,
205}
206
207impl KeyBinding {
208    /// Check if this binding is valid (active and not expired)
209    pub fn is_valid(&self) -> bool {
210        if !self.active {
211            return false;
212        }
213        if let Some(expires) = self.expires_at {
214            if Utc::now() > expires {
215                return false;
216            }
217        }
218        true
219    }
220}

226/// Types of key lifecycle events.
227#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
228#[serde(rename_all = "snake_case")]
229pub enum AuditEventType {
230    /// A new key binding was registered
231    KeyRegistered,
232    /// A key binding was rotated (new key, same identity)
233    KeyRotated,
234    /// A key binding was revoked/deactivated
235    KeyRevoked,
236    /// A key binding expired
237    KeyExpired,
238    /// An approval was granted using this key
239    ApprovalGranted,
240    /// An approval verification succeeded
241    ApprovalVerified,
242    /// An approval verification failed
243    ApprovalFailed,
244    /// Provider was registered
245    ProviderRegistered,
246    /// Provider was removed
247    ProviderRemoved,
248}

250/// An audit event for key lifecycle operations.
251///
252/// These events should be persisted to an audit log for compliance
253/// and forensic analysis.
254#[derive(Debug, Clone, Serialize, Deserialize)]
255pub struct AuditEvent {
256    /// Unique event ID
257    pub id: String,
258
259    /// Event type
260    pub event_type: AuditEventType,
261
262    /// When this event occurred
263    pub timestamp: DateTime<Utc>,
264
265    /// Which provider this relates to
266    pub provider: String,
267
268    /// External identity involved (if applicable)
269    #[serde(skip_serializing_if = "Option::is_none")]
270    pub external_id: Option<String>,
271
272    /// Public key involved (hex-encoded for readability)
273    #[serde(skip_serializing_if = "Option::is_none")]
274    pub public_key_hex: Option<String>,
275
276    /// Who/what triggered this event
277    pub actor: String,
278
279    /// Additional context
280    #[serde(skip_serializing_if = "Option::is_none")]
281    pub details: Option<String>,
282
283    /// Related IDs (warrant_id, approval_id, etc.)
284    #[serde(skip_serializing_if = "Option::is_none")]
285    pub related_ids: Option<Vec<String>>,
286}

318    }

320    /// Add details

322        self.details = Some(details.into());
323        self
324    }

326    /// Add related IDs

330    }

337/// Registry for managing Notaries (identity providers and their key bindings).
338///
339/// The registry handles:
340/// - Provider registration
341/// - Key binding lifecycle (register, rotate, revoke)
342/// - Audit event generation
343///
344/// ## Example
345///
346/// ```rust,ignore
347/// let mut registry = NotaryRegistry::new();
348///
349/// // Register a provider
350/// registry.register_provider("aws-iam", "AWS IAM Identity Provider")?;
351///
352/// // Register a key binding
353/// let binding = KeyBinding {
354///     id: "kb_123".into(),
355///     external_id: "arn:aws:iam::123:user/admin".into(),
356///     provider: "aws-iam".into(),
357///     public_key: admin_key,
358///     // ...
359/// };
360/// registry.register_key(binding, "system")?;
361///
362/// // Resolve identity to key
363/// let key = registry.resolve("aws-iam", "arn:aws:iam::123:user/admin")?;
364///
365/// // Drain audit events for logging
366/// for event in registry.drain_events() {
367///     log::info!("Audit: {:?}", event);
368/// }
369/// ```
370#[derive(Debug, Default)]
371pub struct NotaryRegistry {
372    /// Registered providers: name → description
373    providers: std::collections::HashMap<String, String>,
374
375    /// Key bindings: (provider, external_id) → KeyBinding
376    bindings: std::collections::HashMap<(String, String), KeyBinding>,
377
378    /// Pending audit events (drain these periodically)
379    pending_events: Vec<AuditEvent>,
380}
381
382impl NotaryRegistry {
383    /// Create a new empty registry.
384    pub fn new() -> Self {
385        Self::default()
386    }
387
388    /// Register a new provider.
389    pub fn register_provider(
390        &mut self,
391        name: impl Into<String>,
392        description: impl Into<String>,
393        actor: impl Into<String>,
394    ) {
395        let name = name.into();
396        let actor = actor.into();
397
398        self.providers.insert(name.clone(), description.into());
399
400        self.pending_events.push(
401            AuditEvent::new(AuditEventType::ProviderRegistered, &name, actor)
402                .with_details(format!("Provider '{}' registered", name)),
403        );
404    }
405
406    /// Remove a provider (also removes all its bindings).
407    pub fn remove_provider(&mut self, name: &str, actor: impl Into<String>) {
408        let actor = actor.into();
409
410        // Remove all bindings for this provider
411        let to_remove: Vec<_> = self
412            .bindings
413            .keys()
414            .filter(|(p, _)| p == name)
415            .cloned()
416            .collect();
417
418        for key in to_remove {
419            self.bindings.remove(&key);
420        }
421
422        self.providers.remove(name);
423
424        self.pending_events.push(
425            AuditEvent::new(AuditEventType::ProviderRemoved, name, actor)
426                .with_details(format!("Provider '{}' removed", name)),
427        );
428    }
429
430    /// Check if a provider is registered.
431    pub fn has_provider(&self, name: &str) -> bool {
432        self.providers.contains_key(name)
433    }
434
435    /// List all registered providers.
436    pub fn list_providers(&self) -> Vec<(&str, &str)> {
437        self.providers
438            .iter()
439            .map(|(k, v)| (k.as_str(), v.as_str()))
440            .collect()
441    }
442
443    /// Register a new key binding.
444    pub fn register_key(
445        &mut self,
446        binding: KeyBinding,
447        actor: impl Into<String>,
448    ) -> Result<()> {
449        let actor = actor.into();
450
451        if !self.has_provider(&binding.provider) {
452            return Err(Error::UnknownProvider(binding.provider.clone()));
453        }
454
455        let key = (binding.provider.clone(), binding.external_id.clone());
456
457        self.pending_events.push(
458            AuditEvent::new(AuditEventType::KeyRegistered, &binding.provider, &actor)
459                .with_identity(&binding.external_id)
460                .with_key(&binding.public_key)
461                .with_details(format!(
462                    "Key registered for '{}'",
463                    binding.display_name.as_deref().unwrap_or(&binding.external_id)
464                )),
465        );
466
467        self.bindings.insert(key, binding);
468        Ok(())
469    }
470
471    /// Rotate a key (update the public key for an existing binding).
472    pub fn rotate_key(
473        &mut self,
474        provider: &str,
475        external_id: &str,
476        new_key: PublicKey,
477        actor: impl Into<String>,
478    ) -> Result<()> {
479        let actor = actor.into();
480        let key = (provider.to_string(), external_id.to_string());
481
482        let binding = self.bindings.get_mut(&key).ok_or_else(|| {
483            Error::UnknownProvider(format!("{}:{}", provider, external_id))
484        })?;
485
486        let old_key_hex = hex::encode(binding.public_key.to_bytes());
487
488        binding.public_key = new_key.clone();
489
490        self.pending_events.push(
491            AuditEvent::new(AuditEventType::KeyRotated, provider, actor)
492                .with_identity(external_id)
493                .with_key(&new_key)
494                .with_details(format!("Key rotated from {}", &old_key_hex[..16])),
495        );
496
497        Ok(())
498    }
499
500    /// Revoke a key binding (deactivate it).
501    pub fn revoke_key(
502        &mut self,
503        provider: &str,
504        external_id: &str,
505        reason: impl Into<String>,
506        actor: impl Into<String>,
507    ) -> Result<()> {
508        let actor = actor.into();
509        let reason = reason.into();
510        let key = (provider.to_string(), external_id.to_string());
511
512        let binding = self.bindings.get_mut(&key).ok_or_else(|| {
513            Error::UnknownProvider(format!("{}:{}", provider, external_id))
514        })?;
515
516        binding.active = false;
517
518        self.pending_events.push(
519            AuditEvent::new(AuditEventType::KeyRevoked, provider, actor)
520                .with_identity(external_id)
521                .with_key(&binding.public_key)
522                .with_details(format!("Revoked: {}", reason)),
523        );
524
525        Ok(())
526    }
527
528    /// Resolve an external identity to a public key.
529    pub fn resolve(&self, provider: &str, external_id: &str) -> Result<&PublicKey> {
530        let key = (provider.to_string(), external_id.to_string());
531
532        let binding = self.bindings.get(&key).ok_or_else(|| {
533            Error::UnknownProvider(format!("{}:{}", provider, external_id))
534        })?;
535
536        if !binding.is_valid() {
537            return Err(Error::ApprovalExpired {
538                approved_at: binding.registered_at,
539                expired_at: binding.expires_at.unwrap_or(Utc::now()),
540            });
541        }
542
543        Ok(&binding.public_key)
544    }
545
546    /// Get a key binding by identity.
547    pub fn get_binding(&self, provider: &str, external_id: &str) -> Option<&KeyBinding> {
548        let key = (provider.to_string(), external_id.to_string());
549        self.bindings.get(&key)
550    }
551
552    /// List all bindings for a provider.
553    pub fn list_bindings(&self, provider: &str) -> Vec<&KeyBinding> {
554        self.bindings
555            .iter()
556            .filter(|((p, _), _)| p == provider)
557            .map(|(_, b)| b)
558            .collect()
559    }
560
561    /// Record an approval event (for audit trail).
562    pub fn record_approval(
563        &mut self,
564        event_type: AuditEventType,
565        approval: &Approval,
566        actor: impl Into<String>,
567        details: Option<String>,
568    ) {
569        let mut event = AuditEvent::new(event_type, &approval.provider, actor)
570            .with_identity(&approval.external_id)
571            .with_key(&approval.approver_key);
572
573        if let Some(d) = details {
574            event = event.with_details(d);
575        }
576
577        self.pending_events.push(event);
578    }
579
580    /// Drain pending audit events.
581    ///
582    /// Call this periodically to persist events to your audit log.
583    pub fn drain_events(&mut self) -> Vec<AuditEvent> {
584        std::mem::take(&mut self.pending_events)
585    }
586
587    /// Get pending event count (for monitoring).
588    pub fn pending_event_count(&self) -> usize {
589        self.pending_events.len()
590    }
591}

619        );

622        let hash2 = compute_request_hash("wrt_123", "delete", &args2, None);

630    }

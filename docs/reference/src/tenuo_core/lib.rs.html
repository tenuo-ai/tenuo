1//! # Tenuo Core
2//!
3//! Agent Capability Flow Control - Rust core library.
4//!
5//! Tenuo provides cryptographically-enforced capability attenuation for AI agent workflows.
6//! Unlike traditional IAM systems that answer "Who are you?", Tenuo answers
7//! "Does this actor hold a valid, scoped, unexpired token for this specific action?"
8//!
9//! ## Key Concepts
10//!
11//! - **Warrant**: A transferable token of authority containing tool name, constraints, TTL, and signature
12//! - **Attenuation**: Capabilities can only shrink when delegated, never expand
13//! - **Constraints**: Restrictions on argument values (Pattern, Exact, OneOf, Range, CEL)
14//!
15//! ## Example
16//!
17//! ```rust,ignore
18//! use tenuo_core::{Warrant, Pattern, Keypair};
19//! use std::time::Duration;
20//!
21//! // Create a keypair for the control plane
22//! let keypair = Keypair::generate();
23//!
24//! // Issue a warrant for cluster upgrades
25//! let warrant = Warrant::builder()
26//!     .tool("upgrade_cluster")
27//!     .constraint("cluster", Pattern::new("staging-*"))
28//!     .ttl(Duration::from_secs(600))
29//!     .build(&keypair)?;
30//!
31//! // Attenuate to a narrower scope for a worker agent
32//! let worker_warrant = warrant.attenuate()
33//!     .constraint("cluster", Exact::new("staging-web"))
34//!     .build(&worker_keypair)?;
35//! ```

66/// Maximum delegation depth to prevent unbounded chains (protocol-level hard cap).
67/// 
68/// Individual warrants can set a lower limit via `max_depth` in the payload.
69/// This constant prevents DoS attacks from extremely deep chains.
70pub const MAX_DELEGATION_DEPTH: u32 = 64;

72/// Context string for Ed25519 signatures (prevents cross-protocol attacks).
73/// 
74/// All signatures are computed over: `SIGNATURE_CONTEXT || payload`
75/// 
76/// This prevents a signature from one protocol being valid in another.
77pub const SIGNATURE_CONTEXT: &[u8] = b"tenuo-warrant-v1";

82#[cfg(test)]
83mod tests {
84    use super::*;
85    use std::time::Duration;
86
87    #[test]
88    fn test_basic_warrant_creation() {
89        let keypair = Keypair::generate();
90        
91        let warrant = Warrant::builder()
92            .tool("upgrade_cluster")
93            .constraint("cluster", Pattern::new("staging-*").unwrap())
94            .ttl(Duration::from_secs(600))
95            .build(&keypair)
96            .unwrap();
97
98        assert_eq!(warrant.tool(), "upgrade_cluster");
99        assert!(warrant.verify(&keypair.public_key()).is_ok());
100    }
101
102    #[test]
103    fn test_attenuation_narrows_constraints() {
104        let keypair = Keypair::generate();
105        let child_keypair = Keypair::generate();
106
107        let parent = Warrant::builder()
108            .tool("upgrade_cluster")
109            .constraint("cluster", Pattern::new("staging-*").unwrap())
110            .ttl(Duration::from_secs(600))
111            .build(&keypair)
112            .unwrap();
113
114        let child = parent
115            .attenuate()
116            .constraint("cluster", Exact::new("staging-web"))
117            .build(&child_keypair)
118            .unwrap();
119
120        assert!(child.expires_at() <= parent.expires_at());
121        assert_eq!(child.depth(), parent.depth() + 1);
122    }

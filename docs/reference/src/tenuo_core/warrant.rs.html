1//! Warrant type - the core primitive of Tenuo.
2//!
3//! A warrant is a transferable token of authority containing:
4//! - A tool name (what action is authorized)
5//! - Constraints on arguments (how arguments must be restricted)
6//! - A time-to-live (when the warrant expires)
7//! - A signature proving provenance (who issued it)
8//!
9//! Crucially, capabilities can only shrink when warrants are delegated, never expand.
10

21/// The required prefix for all warrant IDs.
22pub const WARRANT_ID_PREFIX: &str = "tnu_wrt_";
23
24/// Size of the timestamp window for PoP signatures in seconds.
25///
26/// The verifier accepts signatures from 4 consecutive windows (current Â± 2),
27/// giving approximately 2 minutes of tolerance for clock skew.
28///
29/// **Security trade-off**: Larger windows tolerate more clock skew but increase
30/// the replay window. Smaller windows reduce replay risk but may cause false
31/// rejections in distributed systems.
32pub const POP_TIMESTAMP_WINDOW_SECS: i64 = 30;
33
34/// A unique identifier for a warrant.
35/// 
36/// Uses UUIDv7 (time-ordered) which provides:
37/// - 48 bits of millisecond timestamp
38/// - 74 bits of random data
39/// - Monotonically increasing within the same millisecond
40/// - Collision probability: 1 in 2^74 per millisecond (effectively zero)
41/// 
42/// **Validation**: IDs must start with `tnu_wrt_` prefix. This is enforced
43/// during both construction (`from_string`) and deserialization.
44#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize)]
45pub struct WarrantId(String);
46
47impl<'de> Deserialize<'de> for WarrantId {
48    fn deserialize<D>(deserializer: D) -> std::result::Result<Self, D::Error>
49    where
50        D: Deserializer<'de>,
51    {
52        let s = String::deserialize(deserializer)?;
53        if !s.starts_with(WARRANT_ID_PREFIX) {
54            return Err(serde::de::Error::custom(format!(
55                "warrant ID must start with '{}', got: {}",
56                WARRANT_ID_PREFIX,
57                if s.len() > 20 { &s[..20] } else { &s }
58            )));
59        }
60        Ok(WarrantId(s))
61    }
62}

65    /// Generate a new time-ordered warrant ID (UUIDv7).
66    /// 
67    /// UUIDv7 provides both uniqueness and chronological ordering,
68    /// making it ideal for debugging and audit trails.

70        Self(format!("tnu_wrt_{}", Uuid::now_v7().simple()))
71    }

75    /// Use this when you don't want IDs to reveal timing information.

77        Self(format!("tnu_wrt_{}", Uuid::new_v4().simple()))

81    /// 
82    /// Returns `InvalidWarrantId` if the string doesn't start with `tnu_wrt_`.

84        let s = s.into();
85        if !s.starts_with(WARRANT_ID_PREFIX) {
86            return Err(Error::InvalidWarrantId(format!(
87                "warrant ID must start with '{}', got: {}",
88                WARRANT_ID_PREFIX,
89                if s.len() > 20 { &s[..20] } else { &s }
90            )));
91        }
92        Ok(Self(s))
93    }

113/// The payload of a warrant (unsigned).
114#[derive(Debug, Clone, Serialize, Deserialize)]
115pub struct WarrantPayload {
116    /// Unique identifier for this warrant.
117    pub id: WarrantId,
118    /// The tool this warrant authorizes.
119    pub tool: String,
120    /// Constraints on tool arguments.
121    pub constraints: ConstraintSet,
122    /// When this warrant expires.
123    pub expires_at: DateTime<Utc>,
124    /// Delegation depth (0 for root warrants).
125    pub depth: u32,
126    /// Maximum delegation depth allowed from this warrant chain.
127    /// 
128    /// If set, child warrants cannot exceed this depth. This is a policy limit
129    /// that can be set by the Control Plane or any delegator. The value can only
130    /// shrink during attenuation (monotonicity).
131    /// 
132    /// If `None`, the protocol-level `MAX_DELEGATION_DEPTH` applies.
133    #[serde(skip_serializing_if = "Option::is_none")]
134    pub max_depth: Option<u32>,
135    /// Parent warrant ID (None for root warrants).
136    pub parent_id: Option<WarrantId>,
137    /// Session ID for session binding.
138    pub session_id: Option<String>,
139    /// Agent ID for traceability (e.g., UUID of the agent software/instance).
140    pub agent_id: Option<String>,
141    /// Public key of the issuer (who signed this warrant).
142    pub issuer: PublicKey,
143    /// Public key of the authorized holder (Proof-of-Possession).
144    /// 
145    /// If set, the holder must prove they control this key when using the warrant.
146    /// This prevents stolen warrants from being used by attackers.
147    /// 
148    /// When `None`, the warrant is a bearer token (anyone with it can use it).
149    #[serde(skip_serializing_if = "Option::is_none")]
150    pub authorized_holder: Option<PublicKey>,
151
152    /// Public keys of required approvers for multi-sig workflows.
153    /// 
154    /// If set, actions require signatures from approvers in this list.
155    /// Use with `min_approvals` for M-of-N schemes (e.g., 2-of-3).
156    /// 
157    /// When `None`, no additional approvals are required.
158    #[serde(skip_serializing_if = "Option::is_none")]
159    pub required_approvers: Option<Vec<PublicKey>>,
160
161    /// Minimum number of approvals required (for M-of-N multi-sig).
162    /// 
163    /// If `None` but `required_approvers` is set, ALL approvers must sign.
164    /// Must be <= len(required_approvers).
165    #[serde(skip_serializing_if = "Option::is_none")]
166    pub min_approvals: Option<u32>,
167}

169/// A signed warrant - the complete token of authority.
170/// 
171/// **Security**: Custom deserialization validates constraint depth to prevent
172/// stack overflow attacks from maliciously nested constraints.
173#[derive(Debug, Clone, Serialize)]
174pub struct Warrant {
175    /// The warrant payload.
176    payload: WarrantPayload,
177    /// Signature over the payload.
178    signature: Signature,
179    /// Original serialized payload bytes (for deterministic verification).
180    /// This ensures signature verification works even when HashMap serialization
181    /// order varies between serialize/deserialize cycles.
182    #[serde(with = "serde_bytes")]
183    payload_bytes: Vec<u8>,
184}
185
186// Custom Deserialize to enforce constraint depth validation
187impl<'de> Deserialize<'de> for Warrant {
188    fn deserialize<D>(deserializer: D) -> std::result::Result<Self, D::Error>
189    where
190        D: serde::Deserializer<'de>,
191    {
192        // Helper struct for raw deserialization
193        #[derive(Deserialize)]
194        struct WarrantRaw {
195            payload: WarrantPayload,
196            signature: Signature,
197            #[serde(with = "serde_bytes")]
198            payload_bytes: Vec<u8>,
199        }
200
201        let raw = WarrantRaw::deserialize(deserializer)?;
202        
203        // Validate constraint depth before returning
204        raw.payload.constraints.validate_depth()
205            .map_err(serde::de::Error::custom)?;
206        
207        Ok(Warrant {
208            payload: raw.payload,
209            signature: raw.signature,
210            payload_bytes: raw.payload_bytes,
211        })
212    }
213}

314    /// Get the effective approval threshold.
315    /// 
316    /// Returns the number of approvals needed:
317    /// - If `min_approvals` is set, returns that value
318    /// - If `required_approvers` is set but `min_approvals` is not, returns the count (all must sign)

321        match (&self.payload.required_approvers, self.payload.min_approvals) {
322            (Some(approvers), Some(min)) => min.min(approvers.len() as u32),
323            (Some(approvers), None) => approvers.len() as u32, // All must sign
324            (None, _) => 0, // No multi-sig required
325        }

328    /// Verify that a holder signature proves possession.
329    /// 
330    /// The holder must sign a challenge (typically the action being performed)
331    /// to prove they control the `authorized_holder` key.
332    /// 
333    /// # Arguments
334    /// * `challenge` - The data that was signed (e.g., action + timestamp + nonce)
335    /// * `signature` - The holder's signature over the challenge
336    /// 
337    /// # Returns
338    /// * `Ok(())` if no PoP required OR signature is valid
339    /// * `Err` if PoP required but signature invalid or missing holder
340    pub fn verify_holder(&self, challenge: &[u8], signature: &Signature) -> Result<()> {
341        match &self.payload.authorized_holder {
342            None => Ok(()), // Bearer token, no PoP required
343            Some(holder_key) => {
344                holder_key.verify(challenge, signature)
345                    .map_err(|_| Error::SignatureInvalid(
346                        "holder proof-of-possession failed".to_string()
347                    ))
348            }
349        }
350    }

369    /// Verify the warrant signature.
370    pub fn verify(&self, expected_issuer: &PublicKey) -> Result<()> {
371        if &self.payload.issuer != expected_issuer {
372            return Err(Error::SignatureInvalid(
373                "issuer public key does not match".to_string(),
374            ));
375        }
376
377        // Use the stored payload bytes for verification (ensures determinism)
378        self.payload.issuer.verify(&self.payload_bytes, &self.signature)
379    }

381    /// Authorize an action against this warrant.
382    ///
383    /// Checks:
384    /// 1. Warrant is not expired
385    /// 2. Tool name matches
386    /// 3. All constraints are satisfied
387    /// 4. Proof-of-Possession (if authorized_holder is set)
388    pub fn authorize(
389        &self,
390        tool: &str,
391        args: &HashMap<String, ConstraintValue>,
392        signature: Option<&Signature>,
393    ) -> Result<()> {
394        // Check expiration
395        if self.is_expired() {
396            return Err(Error::WarrantExpired(self.payload.expires_at));
397        }
398
399        // Check tool name
400        if self.payload.tool != tool {
401            return Err(Error::ConstraintNotSatisfied {
402                field: "tool".to_string(),
403                reason: format!(
404                    "warrant is for tool '{}', not '{}'",
405                    self.payload.tool, tool
406                ),
407            });
408        }
409
410        // Check constraints
411        self.payload.constraints.matches(args)?;
412
413        // Check Proof-of-Possession
414        if let Some(holder_key) = &self.payload.authorized_holder {
415            let signature = signature.ok_or_else(|| {
416                Error::MissingSignature("Proof-of-Possession required".to_string())
417            })?;
418
419            // PoP signature covers (warrant_id, tool, sorted_args, timestamp_window)
420            // We verify against multiple recent windows to handle clock skew.
421            // 
422            // Security: This creates a ~2 minute replay window. Mitigate with:
423            // - Short-lived warrants (TTL < 2 min)
424            // - Application-layer request deduplication
425            let now = Utc::now().timestamp();
426            let max_windows = 4; // Accept signatures from last ~2 minutes
427            
428            let mut sorted_args: Vec<(&String, &ConstraintValue)> = args.iter().collect();
429            sorted_args.sort_by_key(|(k, _)| *k);
430
431            let mut verified = false;
432            for i in 0..max_windows {
433                // Try current and recent time windows
434                let window_ts = (now / POP_TIMESTAMP_WINDOW_SECS - i) * POP_TIMESTAMP_WINDOW_SECS;
435                let challenge_data = (self.payload.id.as_str(), tool, &sorted_args, window_ts);
436                let mut challenge_bytes = Vec::new();
437                if ciborium::ser::into_writer(&challenge_data, &mut challenge_bytes).is_err() {
438                    continue;
439                }
440                if holder_key.verify(&challenge_bytes, signature).is_ok() {
441                    verified = true;
442                    break;
443                }
444            }
445            
446            if !verified {
447                return Err(Error::SignatureInvalid("Proof-of-Possession failed or expired".to_string()));
448            }
449        }
450
451        Ok(())
452    }

454    /// Create a Proof-of-Possession signature for a request.
455    ///
456    /// This generates a time-bounded signature that proves you hold the private key
457    /// corresponding to the warrant's `authorized_holder`. The signature is valid
458    /// for approximately 2 minutes (to handle clock skew in distributed systems).
459    ///
460    /// The challenge includes: `(warrant_id, tool, sorted_args, timestamp_window)`
461    ///
462    /// # Security Note
463    ///
464    /// **Residual replay risk**: Within the ~2 minute window, a captured PoP signature
465    /// can be replayed. This is an intentional trade-off to handle clock skew.
466    ///
467    /// **Mitigations**:
468    /// - Use short-lived warrants (TTL < 2 min) for high-security operations
469    /// - Implement request deduplication at the application layer using
470    ///   `(warrant_id, tool, args)` as a cache key with 2-minute TTL
471    pub fn create_pop_signature(
472        &self,
473        keypair: &Keypair,
474        tool: &str,
475        args: &HashMap<String, ConstraintValue>,
476    ) -> Result<Signature> {
477        // Create challenge: (warrant_id, tool, sorted_args, timestamp_window)
478        let mut sorted_args: Vec<(&String, &ConstraintValue)> = args.iter().collect();
479        sorted_args.sort_by_key(|(k, _)| *k);
480
481        // Time-bound the signature to a window for replay protection
482        let now = Utc::now().timestamp();
483        let window_ts = (now / POP_TIMESTAMP_WINDOW_SECS) * POP_TIMESTAMP_WINDOW_SECS;
484
485        let challenge_data = (self.payload.id.as_str(), tool, sorted_args, window_ts);
486        let mut challenge_bytes = Vec::new();
487        ciborium::ser::into_writer(&challenge_data, &mut challenge_bytes)
488            .map_err(|e| Error::SerializationError(e.to_string()))?;
489
490        Ok(keypair.sign(&challenge_bytes))
491    }

589        self.min_approvals = Some(min);
590        self
591    }
592
593    /// Build and sign the warrant.
594    pub fn build(self, keypair: &Keypair) -> Result<Warrant> {
595        let tool = self.tool.ok_or(Error::MissingField("tool".to_string()))?;
596        let ttl = self.ttl.ok_or(Error::MissingField("ttl".to_string()))?;
597
598        // Validate max_depth doesn't exceed protocol cap
599        if let Some(max) = self.max_depth {
600            if max > MAX_DELEGATION_DEPTH {
601                return Err(Error::DepthExceeded(max, MAX_DELEGATION_DEPTH));
602            }
603        }
604
605        let chrono_ttl = ChronoDuration::from_std(ttl)
606            .map_err(|_| Error::InvalidTtl("TTL too large".to_string()))?;
607        let expires_at = Utc::now() + chrono_ttl;
608
609        // Validate min_approvals if set
610        if let (Some(approvers), Some(min)) = (&self.required_approvers, self.min_approvals) {
611            if min as usize > approvers.len() {
612                return Err(Error::MonotonicityViolation(format!(
613                    "min_approvals ({}) cannot exceed required_approvers count ({})",
614                    min, approvers.len()
615                )));
616            }
617        }
618
619        let payload = WarrantPayload {
620            id: WarrantId::new(),
621            tool,
622            constraints: self.constraints,
623            expires_at,
624            depth: 0,
625            max_depth: self.max_depth,
626            parent_id: None,
627            session_id: self.session_id,
628            agent_id: self.agent_id,
629            issuer: keypair.public_key(),
630            authorized_holder: self.authorized_holder,
631            required_approvers: self.required_approvers,
632            min_approvals: self.min_approvals,
633        };
634
635        let mut payload_bytes = Vec::new();
636        ciborium::ser::into_writer(&payload, &mut payload_bytes)?;
637        let signature = keypair.sign(&payload_bytes);
638
639        Ok(Warrant { payload, signature, payload_bytes })
640    }

674            // Multi-sig: inherit from parent (can only add MORE approvers or raise threshold)
675            required_approvers: parent.payload.required_approvers.clone(),
676            min_approvals: parent.payload.min_approvals,
677        }
678    }

680    /// Override a constraint with a narrower one.
681    pub fn constraint(mut self, field: impl Into<String>, constraint: impl Into<Constraint>) -> Self {
682        self.constraints.insert(field, constraint);
683        self
684    }

692    /// Set a lower maximum delegation depth.
693    /// 
694    /// This can only shrink the parent's `max_depth`, never expand it.

704    /// Set or change the agent ID.
705    pub fn agent_id(mut self, agent_id: impl Into<String>) -> Self {
706        self.agent_id = Some(agent_id.into());
707        self

710    /// Set or change the authorized holder (Proof-of-Possession).
711    pub fn authorized_holder(mut self, public_key: PublicKey) -> Self {
712        self.authorized_holder = Some(public_key);
713        self
714    }

716    /// Add required approvers (can only add more, not remove).
717    /// 
718    /// Multi-sig is monotonic: you can add approvers but not remove them.
719    /// The new approvers are merged with any inherited from the parent.
720    pub fn add_approvers(mut self, approvers: Vec<PublicKey>) -> Self {
721        let mut current = self.required_approvers.unwrap_or_default();
722        for approver in approvers {
723            if !current.contains(&approver) {
724                current.push(approver);
725            }
726        }
727        self.required_approvers = Some(current);
728        self
729    }
730
731    /// Increase the minimum approvals required (can only increase).
732    /// 
733    /// Multi-sig threshold is monotonic: you can raise it but not lower it.
734    pub fn raise_min_approvals(mut self, min: u32) -> Self {
735        let current = self.min_approvals.unwrap_or(0);
736        self.min_approvals = Some(min.max(current));
737        self
738    }
739
740    /// Validate multi-sig monotonicity (cannot remove approvers or lower threshold).
741    fn validate_multisig_monotonicity(&self) -> Result<()> {
742        // 1. Cannot remove approvers (child must include all parent approvers)
743        if let Some(parent_approvers) = &self.parent.payload.required_approvers {
744            if let Some(child_approvers) = &self.required_approvers {
745                for parent_key in parent_approvers {
746                    if !child_approvers.contains(parent_key) {
747                        return Err(Error::MonotonicityViolation(format!(
748                            "cannot remove approver {} from multi-sig set",
749                            hex::encode(parent_key.to_bytes())
750                        )));
751                    }
752                }
753            } else {
754                // Child doesn't have approvers but parent does - violation
755                return Err(Error::MonotonicityViolation(
756                    "cannot remove multi-sig requirement from parent".to_string()
757                ));
758            }
759        }
760
761        // 2. Cannot lower min_approvals
762        if let Some(parent_min) = self.parent.payload.min_approvals {
763            if let Some(child_min) = self.min_approvals {
764                if child_min < parent_min {
765                    return Err(Error::MonotonicityViolation(format!(
766                        "cannot lower min_approvals from {} to {}",
767                        parent_min, child_min
768                    )));
769                }
770            }
771            // If parent has min_approvals but child doesn't set it, inherit (ok)
772        }
773
774        // 3. Validate min_approvals doesn't exceed approvers count
775        if let (Some(approvers), Some(min)) = (&self.required_approvers, self.min_approvals) {
776            if min as usize > approvers.len() {
777                return Err(Error::MonotonicityViolation(format!(
778                    "min_approvals ({}) cannot exceed required_approvers count ({})",
779                    min, approvers.len()
780                )));
781            }
782        }
783
784        Ok(())
785    }
786
787    /// Build and sign the attenuated warrant.
788    pub fn build(self, keypair: &Keypair) -> Result<Warrant> {
789        // Use checked arithmetic to prevent overflow
790        let new_depth = self.parent.payload.depth
791            .checked_add(1)
792            .ok_or(Error::DepthExceeded(u32::MAX, MAX_DELEGATION_DEPTH))?;
793        
794        // Calculate effective max_depth (monotonic: can only shrink)
795        let effective_max = match (self.parent.payload.max_depth, self.max_depth) {
796            // Both set: take the minimum (can only shrink)
797            (Some(parent_max), Some(child_max)) => {
798                if child_max > parent_max {
799                    return Err(Error::MonotonicityViolation(format!(
800                        "max_depth {} exceeds parent's max_depth {}",
801                        child_max, parent_max
802                    )));
803                }
804                Some(child_max)
805            }
806            // Parent set, child not: inherit parent's limit
807            (Some(parent_max), None) => Some(parent_max),
808            // Child set, parent not: use child's limit (capped by protocol)
809            (None, Some(child_max)) => {
810                if child_max > MAX_DELEGATION_DEPTH {
811                    return Err(Error::DepthExceeded(child_max, MAX_DELEGATION_DEPTH));
812                }
813                Some(child_max)
814            }
815            // Neither set: no limit (protocol default applies)
816            (None, None) => None,
817        };
818
819        // Check depth against effective limit
820        let depth_limit = effective_max.unwrap_or(MAX_DELEGATION_DEPTH);
821        if new_depth > depth_limit {
822            return Err(Error::DepthExceeded(new_depth, depth_limit));
823        }
824
825        // Also enforce protocol hard cap
826        if new_depth > MAX_DELEGATION_DEPTH {
827            return Err(Error::DepthExceeded(new_depth, MAX_DELEGATION_DEPTH));
828        }
829
830        // Check parent is not expired
831        if self.parent.is_expired() {
832            return Err(Error::WarrantExpired(self.parent.payload.expires_at));
833        }
834
835        // Validate attenuation monotonicity for constraints
836        self.parent.payload.constraints.validate_attenuation(&self.constraints)?;
837
838        // Validate multi-sig monotonicity (cannot remove approvers or lower threshold)
839        self.validate_multisig_monotonicity()?;
840
841        // Calculate expiration (must not exceed parent)
842        let expires_at = if let Some(ttl) = self.ttl {
843            let chrono_ttl = ChronoDuration::from_std(ttl)
844                .map_err(|_| Error::InvalidTtl("TTL too large".to_string()))?;
845            let proposed = Utc::now() + chrono_ttl;
846            if proposed > self.parent.payload.expires_at {
847                self.parent.payload.expires_at
848            } else {
849                proposed
850            }
851        } else {
852            self.parent.payload.expires_at
853        };
854
855        // Determine effective min_approvals (inherit if not set)
856        let effective_min = self.min_approvals.or(self.parent.payload.min_approvals);
857
858        let payload = WarrantPayload {
859            id: WarrantId::new(),
860            tool: self.parent.payload.tool.clone(),
861            constraints: self.constraints,
862            expires_at,
863            depth: new_depth,
864            max_depth: effective_max,
865            parent_id: Some(self.parent.payload.id.clone()),
866            session_id: self.session_id,
867            agent_id: self.agent_id,
868            issuer: keypair.public_key(),
869            authorized_holder: self.authorized_holder,
870            required_approvers: self.required_approvers,
871            min_approvals: effective_min,
872        };
873
874        let mut payload_bytes = Vec::new();
875        ciborium::ser::into_writer(&payload, &mut payload_bytes)?;
876        let signature = keypair.sign(&payload_bytes);
877
878        Ok(Warrant { payload, signature, payload_bytes })
879    }

# This runs once before the main container starts
# Feature: Fetch initial warrant before the main app starts to ensure secure-by-default.

import os
import requests
from tenuo import Warrant, Pattern

# Authenticate with Control Plane using service account token
# ENV VARIABLE: CONTROL_PLANE_URL, AGENT_ID
control_url = os.getenv("CONTROL_PLANE_URL")
agent_id = os.getenv("AGENT_ID")

# Request warrant for this pod
# SIMULATION: In production, this would use proper authentication
response = requests.post(
    f"{control_url}/v1/warrants",
    json={"agent_id": agent_id, "constraints": {"file_path": "/tmp/*"}}
)
warrant_b64 = response.json()["warrant_base64"]

# Write to shared volume for main container
# HARDCODED PATH: /etc/tenuo/warrant.b64 (mounted from emptyDir)
# In production: Use env var TENUO_WARRANT_FILE_PATH
with open("/etc/tenuo/warrant.b64", "w") as f:
    f.write(warrant_b64)

# Sidecar: Validate per-request warrants (OFFLINE)
# Optional - only needed for request-level authorization
# The sidecar intercepts requests and validates warrants before forwarding to the main container.
# All verification happens offline - no network calls to the control plane.

from tenuo import Warrant, Authorizer, PublicKey
from tenuo.exceptions import AuthorizationError
import os

# Load trusted public key at startup (from K8s Secret)
# ENV VARIABLE: TENUO_TRUSTED_PUBLIC_KEY (hex-encoded)
trusted_key_hex = os.getenv("TENUO_TRUSTED_PUBLIC_KEY")
trusted_key = PublicKey.from_bytes(bytes.fromhex(trusted_key_hex))
authorizer = Authorizer.new(trusted_key)

# For each incoming request
def validate_request(request):
    # Extract warrant from header (set by ingress/API gateway)
    warrant_b64 = request.headers.get("X-Tenuo-Warrant")
    if not warrant_b64:
        return Response("No warrant", status_code=403)
    
    # Parse and verify warrant (100% offline, ~25Œºs)
    warrant = Warrant.from_base64(warrant_b64)
    try:
        # Extract tool and args from request
        tool = request.path.split('/')[-1]
        args = request.json()
        authorizer.check(warrant, tool, args)
        # ‚úì Valid - forward to main container
        return forward_to_main(request)
    except AuthorizationError:
        # ‚úó Invalid - return 403
        return Response("Unauthorized", status_code=403)

Context View: Async State Management
üîë Solving the "Invisible State" Problem
‚ùå The Problem

In async Python (FastAPI, LangChain), function calls happen across await boundaries. Passing context (like warrants) through the call stack is hard - you'd need to manually pass it to every function.

‚úÖ The Solution

Tenuo uses Python's ContextVar which automatically propagates through async operations. Set the warrant once in middleware, and all @lockdown functions have access to it - no manual passing required.

1. Request Arrives
HTTP request arrives at FastAPI/LangChain application. Warrant may be in header or loaded from pod config.
ContextVar: None
2. Middleware Sets Context
FastAPI middleware extracts warrant and sets it in ContextVar. This propagates through all async operations. 
# FastAPI middleware
from tenuo import set_warrant_context

@app.middleware("http")
async def tenuo_middleware(request, call_next):
    # Load warrant from header or pod config
    warrant = load_warrant(request)
    
    # Set in ContextVar - propagates through all async operations
    with set_warrant_context(warrant):
        return await call_next(request)
ContextVar: Warrant(id="...")
3. LangChain AgentExecutor
AgentExecutor processes request. ContextVar automatically propagates through await boundaries. No manual passing needed.
ContextVar: Warrant(id="...")
‚úì Propagates through await
4. Tool Function Called
@lockdown decorator or protect_tools() wrapper retrieves warrant from ContextVar, checks authorization, then executes tool function. 
# Option A: @lockdown decorator (FastAPI/Flask apps)
from tenuo import lockdown

@lockdown(tool="read_file",
          extract_args=lambda file_path, **kwargs: {"file_path": file_path})
def read_file(file_path: str) -> str:
    # @lockdown retrieves warrant from ContextVar automatically
    with open(file_path, 'r') as f:
        return f.read()

# Option B: protect_tools() for LangChain (preferred for agents)
from tenuo.langchain import protect_tools
secure_tools = protect_tools([read_file], warrant=warrant, keypair=keypair)

Standard K8s Deployment Pattern
YAML configuration showing init container, volume, and main container
# Standard K8s Deployment - No custom operators needed
# Uses only standard Kubernetes patterns

apiVersion: apps/v1
kind: Deployment
metadata:
  name: langchain-agent
spec:
  template:
    spec:
      # 1. Init container: Fetch warrant from Control Plane (ONLINE)
      # Feature: Fetch initial warrant before the main app starts to ensure secure-by-default.
      # The init container runs once at pod startup, fetches warrant from control plane (online),
      # and writes it to a shared volume. The main container then loads it at startup (offline).
      initContainers:
      - name: tenuo-warrant-fetcher
        image: your-registry/tenuo-warrant-fetcher:latest # Replace with your actual image
        env:
        - name: AGENT_ID
          valueFrom:
            fieldRef:
              fieldPath: metadata.name # Use pod name as agent ID for traceability
        - name: CONTROL_PLANE_URL
          # ENV VARIABLE: CONTROL_PLANE_URL
          value: "http://tenuo-control-plane.tenuo.svc.cluster.local:8080" # K8s service URL
        - name: OUTPUT_PATH
          # HARDCODED: Path for mounted secret
          value: "/etc/tenuo/warrant.b64"
        volumeMounts:
        - name: tenuo-warrant-volume
          mountPath: /etc/tenuo
      
      # 2. Main container: Load warrant from volume (OFFLINE)
      containers:
      - name: agent-app
        image: my-langchain-app:latest
        env:
        - name: OPENAI_API_KEY
          valueFrom:
            secretKeyRef:
              name: openai-key
              key: api_key
        volumeMounts:
        - name: tenuo-warrant-volume
          mountPath: /etc/tenuo
          readOnly: true
      
      # 3. Shared volume (emptyDir for init‚Üímain communication)
      volumes:
      - name: tenuo-warrant-volume
        emptyDir: {}
